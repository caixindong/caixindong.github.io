栈的实现(C++版)

栈

我利用数组来实现栈，实现方法：
1. 创建MAXSIZE大小的数组用于存放数据，MAXSIZE也为栈的最大容量；
2. 当压栈时，判断栈是否满了，如果不满，就往数组里添加数据，如果满了，就提示栈溢出；
3. 当弹栈时，判断栈是否为空，如果不为空，就取出数组末尾的元素，取出后将该位置的值置为NULL，数组指针向前移动一位；如果为 空，就提示栈为空；
代码实现：

#include <iostream>
#define MAXSIZE 4
using namespace std;
template<class T>
class MyStack{
int size;
T* arr;
public:
MyStack(){
    size = 0;
    arr = new T[MAXSIZE];

}

~MyStack(){
    delete [] arr;
}

int sizeOfStack(){
    return size;
}
//压栈
void push(const T &t){
    if (isFull()) {
        cout<<"栈溢出"<<endl;
        exit(1);
    }else{
        arr[size] = t;
        size++;

    }
}
//弹栈
T pop(){
    if (isEmpty()) {
        cout<<"栈为空"<<endl;
        exit(0);
    }else{
        T t = peek();
        arr[size-1]=NULL;
        size--;
        return t;
    }
}
//查看栈顶元素
T peek(){
    if (isEmpty()) {
        cout<<"栈为空"<<endl;
        exit(0);
    }else{
        return arr[size-1];
    }
}
//判断栈是否为空
bool isEmpty(){
    if (size==0) {
        return true;
    }else{
        return false;
    }
}
//判断栈是否溢出
bool isFull(){
    if (size>=MAXSIZE) {
        return true;
    }else{
        return false;
    }
}
};
